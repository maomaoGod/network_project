141 void CmdView::DealEnter()

154  strText.Empty();
155  if (HintLine < num - 1){ //多行命令
156  num = myedit->GetLineCount();
157  for (nline= HintLine + 1; nline< num; nline++){
158  nIndex = myedit->LineIndex(nline);
159  len = myedit->LineLength(nIndex);
160  myedit->GetLine(nline, strText.GetBuffer(len), len);
161  strText.ReleaseBuffer();
162  command += strText;
163  strText.Empty();
164  }
165  }
166  if (THREADFLAG == THREAD_RUN&&CMDFLAG == ROOT)//未接管模式下命令输入过快
167  PrintLog(_T("命令无效：请等待上一命令执行完"));
168  else if (THREADFLAG == THREAD_WAIT) //未接管模式下发送命令
169  THREADFLAG = THREAD_RUN;

171  LINEFLAG = NEWLINE;
